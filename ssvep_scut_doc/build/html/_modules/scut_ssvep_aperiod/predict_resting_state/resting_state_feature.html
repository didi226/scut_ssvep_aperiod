def delete_all_files_in_folder(folder_path):
	try:
		# Check if the folder exists
		if not os.path.exists(folder_path):
			print(f"The folder {folder_path} does not exist.")
			return

		# Loop through all the files and folders in the directory
		for filename in os.listdir(folder_path):
			file_path = os.path.join(folder_path, filename)
			try:
				# Check if it is a file and remove it
				if os.path.isfile(file_path) or os.path.islink(file_path):

def resting_state_feature(form_path, info_path, pro_ica=True, filter_para=None):
	"""
	:param form_path:              str             path of form (subject_id---root_directory---file_name)
	:param info_path:              str             path of info (mat file for infromation to data)
	:param pro_ica:                bool            whether to do ica in propresess
	:param filter_para:            None/list       Default None  no filters
	                                               [low_freq, high_freq]
	:return:
	"""
	info_form = pd.read_excel(form_path)
	unique_subject_ids = info_form['subject_id'].unique()
	acc_all = np.zeros(len(unique_subject_ids))
	for subject_id in unique_subject_ids:
		subject_rows = info_form.loc[info_form['subject_id'] == subject_id]
		root_directory = subject_rows['root_directory'].tolist()[0]
		file_name = subject_rows['file_name'].tolist()[0]
		data_path = os.path.join(root_directory, file_name)
		feature_directory = os.path.join(root_directory, 'feature')
		# delete_all_files_in_folder(feature_directory)
		datasetone = LoadDataLeeOne(data_path,info_path = info_path)
		pre_data_train, post_data_train, pre_data_test, post_data_test = (
			datasetone.get_data_resting_state(pro_ica = pro_ica, filter_para = filter_para, resample = 4))
		data = np.concatenate([pre_data_train[None,:], post_data_train[None,:], pre_data_test[None,:], post_data_test[None,:]])
		if not os.path.exists(feature_directory):
			os.makedirs(feature_directory)
		save_file_path_psd = os.path.join(feature_directory, os.path.splitext(file_name)[0] + "psd.npy")
		fea1 = Feature(data=data, sfreq=250,  selected_funcs = ['pow_freq_bands'],
		               funcs_params = dict({'pow_freq_bands__freq_bands': [[1, 4], [4, 8], [8, 12], [12, 30], [30, 40]],
                            'pow_freq_bands__normalize': False, 'pow_freq_bands__log': True}))
		np.save(save_file_path_psd, fea1.features)
		dict_fre_band = {"gamma_alpha": np.array([[8, 12], [30, 45]]), "gamma_theta": np.array([[4, 8], [30, 45]]),
		                 "beta_theta": np.array([[4, 8], [12, 30]]), "beta_alpha": np.array([[8, 12], [12, 30]])}
		for key in dict_fre_band:
			value = dict_fre_band[key]
			save_file_path_pac = os.path.join(feature_directory, os.path.splitext(file_name)[0] + key + "pac.npy")
			fea2 = Feature(data=data, sfreq=250,
			               selected_funcs=['pac_connectivity'],
			               funcs_params=dict({'pac_connectivity__sfreq': 250,
			                                 'pac_connectivity__band': value,
			                                 'pac_connectivity__approach_pac': "max"}))
			np.save(save_file_path_pac, fea2.features)
		fea3 = Feature(data=data, sfreq=250,
		               selected_funcs=['aperiodic_periodic_offset_exponent_cf'],
		               funcs_params=dict({'aperiodic_periodic_offset_exponent_cf__sfreq': 250,
		                                 'aperiodic_periodic_offset_exponent_cf__freq_range': [1,40]}))
		save_file_path_aperiodic = os.path.join(feature_directory, os.path.splitext(file_name)[0] +  "aperiodic.npy")
		np.save(save_file_path_aperiodic, fea3.features)



	return acc_all
