    t_stats = np.zeros((4))
    p_values = np.zeros((4))
    for i, i_class in enumerate([5.45,6.67,8.57,12]):

                                          data[file_name + file_name_last_list[1] + data_type + str(i_class)], alternative='greater')
    print(t_stats,p_values)
    return t_stats,p_values
